#include <iostream>
#include <string>

using namespace std;

class chrctr {
protected:
    string nm;
    int hp;
    int mxhp;
    int mn;

public:
    chrctr() {}

    virtual void act(chrctr& tgt, int choice) {
    }

    string getnm() { return nm; }
    int gethp() { return hp; }
    int getmn() { return mn; }

    bool isalv() {
        if (hp > 0) return true;
        else return false;
    }

    void tkdmg(int d) {
        hp = hp - d;
        if (hp < 0) {
            hp = 0;
        }
        cout << "   -> " << nm << " takes " << d
             << " damage! (HP left: " << hp << ")\n";
    }
};

class war : public chrctr {
public:
    war(string name_input) {
        nm = name_input;
        hp = 150;
        mxhp = 150;
        mn = 0;
    }

    void act(chrctr& tgt, int choice) override {
        if (choice == 1) {
            cout << "\n> " << nm << " uses a Sword Attack!\n";
            tgt.tkdmg(25);
        } else if (choice == 2) {
            cout << "\n> " << nm << " uses a Heavy Smash!\n";
            tgt.tkdmg(40);
        } else {
            cout << "\n> " << nm << " got confused and missed!\n";
        }
    }
};

class mag : public chrctr {
public:
    mag(string name_input) {
        nm = name_input;
        hp = 80;
        mxhp = 80;
        mn = 40;
    }

    void act(chrctr& tgt, int choice) override {
        if (choice == 1) {
            cout << "\n> " << nm << " uses a Staff Bonk!\n";
            tgt.tkdmg(10);
        } else if (choice == 2) {
            if (mn < 20) {
                cout << "\n> " << nm << " fails: Not enough mana!\n";
                return;
            }
            mn = mn - 20;
            cout << "\n> " << nm << " casts a Fireball!\n";
            tgt.tkdmg(40);
        } else {
            cout << "\n> " << nm << " got confused and missed!\n";
        }
    }
};

class enmy : public chrctr {
public:
    enmy(string name_input, int health_input) {
        nm = name_input;
        hp = health_input;
        mxhp = health_input;
        mn = 0;
    }

    void act(chrctr& tgt, int choice) override {
        cout << "\n> " << nm << " bites " << tgt.getnm() << " fiercely!\n";
        tgt.tkdmg(15);
    }
};

int main() {
    war p1("Arthur");
    mag p2("Merlin");
    enmy e1("Goblin Boss", 100);

    cout << "================================\n";
    cout << "      SIM RPG - TERMINAL\n";
    cout << "================================\n";

    int round = 1;
    bool battleActive = true;

    while (battleActive) {
        cout << "\n--- ROUND " << round << " ---\n";

        if (p1.isalv() && e1.isalv()) {
            cout << "\n[" << p1.getnm() << "'s Turn | HP: "
                 << p1.gethp() << "]\n";
            cout << "1. Sword Attack (25 Dmg)\n";
            cout << "2. Heavy Smash (40 Dmg)\n";
            cout << "Enter your choice (1 or 2): ";
            int choice;
            cin >> choice;
            p1.act(e1, choice);
        }

        if (p2.isalv() && e1.isalv()) {
            cout << "\n[" << p2.getnm() << "'s Turn | HP: "
                 << p2.gethp() << " | Mana: " << p2.getmn() << "]\n";
            cout << "1. Staff Bonk (10 Dmg, 0 Mana)\n";
            cout << "2. Fireball (40 Dmg, Costs 20 Mana)\n";
            cout << "Enter your choice (1 or 2): ";
            int choice;
            cin >> choice;
            p2.act(e1, choice);
        }

        if (e1.isalv()) {
            cout << "\n[Enemy Phase]\n";
            if (p1.isalv()) {
                e1.act(p1, 0);
            } else if (p2.isalv()) {
                e1.act(p2, 0);
            }
        }

        if (!p1.isalv() && !p2.isalv()) {
            battleActive = false;
        } else if (!e1.isalv()) {
            battleActive = false;
        }

        round++;
    }

    if (p1.isalv() || p2.isalv()) {
        cout << "\n=== VICTORY! The boss is defeated! ===\n";
    } else {
        cout << "\n=== DEFEAT! The heroes died. ===\n";
    }

    return 0;
}
// Base class — shared data and interface
class chrctr {
protected:
    string nm;    // name
    int hp;       // current hit points
    int mxhp;     // maximum hit points
    int mn;       // mana

public:
    chrctr() {}

    // Virtual act — overridden by each subclass
    virtual void act(chrctr& tgt, int choice) {}

    string getnm() { return nm; }
    int gethp()    { return hp; }
    int getmn()    { return mn; }

    bool isalv() {
        if (hp > 0) return true;
        else return false;
    }

    void tkdmg(int d) {
        hp = hp - d;
        if (hp < 0) { hp = 0; }
        cout << "   -> " << nm << " takes " << d
             << " damage! (HP left: " << hp << ")\n";
    }
};

// Warrior — 150 HP, no mana; choice 1: Sword Attack, choice 2: Heavy Smash
class war : public chrctr {
public:
    war(string name_input) {
        nm = name_input; hp = 150; mxhp = 150; mn = 0;
    }
    void act(chrctr& tgt, int choice) override {
        if      (choice == 1) { cout << "\n> " << nm << " uses a Sword Attack!\n"; tgt.tkdmg(25); }
        else if (choice == 2) { cout << "\n> " << nm << " uses a Heavy Smash!\n";  tgt.tkdmg(40); }
        else                  { cout << "\n> " << nm << " got confused and missed!\n"; }
    }
};

// Mage — 80 HP, 40 mana; choice 1: Staff Bonk, choice 2: Fireball (costs 20 mana)
class mag : public chrctr {
public:
    mag(string name_input) {
        nm = name_input; hp = 80; mxhp = 80; mn = 40;
    }
    void act(chrctr& tgt, int choice) override {
        if (choice == 1) {
            cout << "\n> " << nm << " uses a Staff Bonk!\n";
            tgt.tkdmg(10);
        } else if (choice == 2) {
            if (mn < 20) { cout << "\n> " << nm << " fails: Not enough mana!\n"; return; }
            mn = mn - 20;
            cout << "\n> " << nm << " casts a Fireball!\n";
            tgt.tkdmg(40);
        } else {
            cout << "\n> " << nm << " got confused and missed!\n";
        }
    }
};

// Enemy — configurable HP, no mana; always bites for 15 dmg
class enmy : public chrctr {
public:
    enmy(string name_input, int health_input) {
        nm = name_input; hp = health_input; mxhp = health_input; mn = 0;
    }
    void act(chrctr& tgt, int choice) override {
        cout << "\n> " << nm << " bites " << tgt.getnm() << " fiercely!\n";
        tgt.tkdmg(15);
    }
};

// Base class declares virtual act — subclasses override it
class chrctr {
public:
    virtual void act(chrctr& tgt, int choice) {}
};

// war::act — dispatches Sword Attack or Heavy Smash based on player choice
void war::act(chrctr& tgt, int choice) {
    if      (choice == 1) tgt.tkdmg(25);   // Sword Attack
    else if (choice == 2) tgt.tkdmg(40);   // Heavy Smash
    else cout << nm << " got confused and missed!\n";
}

// mag::act — dispatches Staff Bonk or Fireball based on player choice
void mag::act(chrctr& tgt, int choice) {
    if (choice == 1) { tgt.tkdmg(10); }                  // Staff Bonk
    else if (choice == 2) { mn -= 20; tgt.tkdmg(40); }   // Fireball
}

// enmy::act — always bites; choice parameter is ignored
void enmy::act(chrctr& tgt, int choice) {
    tgt.tkdmg(15);
}

// In main() — same act() call, different runtime behaviour:
war  p1("Arthur");
mag  p2("Merlin");
enmy e1("Goblin Boss", 100);

int choice;
cin >> choice;
p1.act(e1, choice);   // resolves to war::act   at runtime
p2.act(e1, choice);   // resolves to mag::act   at runtime
e1.act(p1, 0);        // resolves to enmy::act  at runtime

THE FOUR OOP PILLARS

01 🔒

ENCAPSULATION

The chrctr class hides its data (nm, hp, mxhp, mn) behind protected visibility. External code can only read stats through controlled getters like gethp(), getnm(), and getmn(), while mutation happens through tkdmg(), which enforces a safe lower limit — HP is clamped to 0 and can never go negative. The constructor assigns each member directly in the subclass body, making it explicit what each value controls. This keeps the object's internal state consistent at all times.

02 🌳

INHERITANCE

All character types derive from the single base class chrctr. Hero classes war and mag, and enemy class enmy, inherit shared logic (tkdmg, isalv, the getter methods) without rewriting it. Each subclass only sets its own starting stats (nm, hp, mxhp, mn) in its constructor body and overrides act() with class-specific behaviour. Shared behaviour lives once in the base class, eliminating code duplication across all three concrete types.

03 ⚡

POLYMORPHISM

The virtual act(chrctr& tgt, int choice) in chrctr enables runtime dispatch: the same p1.act(e1, choice) call fires a warrior's sword strike or heavy smash, while p2.act(e1, choice) fires a staff bonk or fireball — and e1.act(p1, 0) always triggers a fierce bite — all depending on the object's true type. The game loop calls act() uniformly; the compiler dispatches to the correct subclass method at runtime. No type-checking if/switch is needed — this is polymorphism in action.

04 🎯

ABSTRACTION

chrctr defines what every character must be able to do — take damage (tkdmg), report health (isalv, gethp), and perform an action (act) — without dictating how. Each subclass fills in the details: war delivers physical strikes, mag manages a mana pool for spells, and enmy always bites. The main() battle loop operates on chrctr references and never needs to know the specifics of any subclass, making it easy to add new character types in the future.
