#include <iostream>
#include <string>
using namespace std;

class Character {
protected:
    string name;
    int hp, maxHp, mana, speed;
    bool isHero;
public:
    Character(string n, int h, int m, int s, bool hero) {
        name   = n;
        hp     = h;
        maxHp  = h;
        mana   = m;
        speed  = s;
        isHero = hero;
    }
    virtual ~Character() {}
    virtual void act(Character* target) = 0;

    string getName()   { return name;   }
    int    getHp()     { return hp;     }
    int    getMana()   { return mana;   }
    int    getSpeed()  { return speed;  }
    bool   getIsHero() { return isHero; }
    bool   isAlive()   { return hp > 0; }

    void takeDamage(int dmg) { hp -= dmg; if (hp < 0) hp = 0; }
    void heal(int amt)       { hp += amt; if (hp > maxHp) hp = maxHp; }
    void useMana(int amt)    { mana -= amt; if (mana < 0) mana = 0; }
};

class Warrior : public Character {
public:
    Warrior(string n) : Character(n, 150, 20, 10, true) {}
    void act(Character* target) override {
        cout << name << " attacks " << target->getName()
             << " for 25 damage!" << endl;
        target->takeDamage(25);
    }
};

class Mage : public Character {
public:
    Mage(string n) : Character(n, 80, 100, 15, true) {}
    void castFireball(Character* target) {
        if (mana < 30) { cout << name << " has no mana for Fireball!" << endl; return; }
        useMana(30);
        cout << name << " casts Fireball on " << target->getName()
             << " for 40 damage!" << endl;
        target->takeDamage(40);
    }
    void castIceStrike(Character* target) {
        if (mana < 20) { cout << name << " has no mana for Ice Strike!" << endl; return; }
        useMana(20);
        cout << name << " casts Ice Strike on " << target->getName()
             << " for 25 damage!" << endl;
        target->takeDamage(25);
    }
    void act(Character* target) override { castFireball(target); }
};

class Healer : public Character {
public:
    Healer(string n) : Character(n, 90, 120, 14, true) {}
    void healAlly(Character* ally) {
        if (mana < 25) { cout << name << " has no mana to heal!" << endl; return; }
        useMana(25);
        cout << name << " heals " << ally->getName() << " for 40 HP!" << endl;
        ally->heal(40);
    }
    void act(Character* target) override { healAlly(target); }
};

class Enemy : public Character {
public:
    Enemy(string n, int h, int s) : Character(n, h, 0, s, false) {}
    Enemy(const Enemy& e)
        : Character(e.name + " (Clone)", e.hp, e.mana, e.speed, false) {}
    void act(Character* target) override {
        cout << name << " attacks " << target->getName()
             << " for 15 damage!" << endl;
        target->takeDamage(15);
    }
};

int main() {
    Warrior arthur("Arthur");
    Mage    merlin("Merlin");
    Healer  elara("Elara");
    Enemy   goblin("Goblin Scout", 60, 12);
    Enemy   orc("Orc Brute", 120, 8);
    Enemy   goblinClone = goblin;

    Character* heroes[3]  = { &arthur, &merlin, &elara };
    Character* enemies[3] = { &goblin, &orc, &goblinClone };

    cout << "=== RPG BATTLE INITIATED ===" << endl;
    int  round = 1;
    bool battleActive = true;

    while (battleActive) {
        cout << "--- ROUND " << round << " ---" << endl;

        for (int i = 0; i < 3; i++) {
            if (!heroes[i]->isAlive()) continue;
            if (i == 2) {
                elara.healAlly(heroes[0]);
            } else {
                for (int j = 0; j < 3; j++) {
                    if (enemies[j]->isAlive()) {
                        heroes[i]->act(enemies[j]);
                        break;
                    }
                }
            }
        }

        for (int i = 0; i < 3; i++) {
            if (!enemies[i]->isAlive()) continue;
            for (int j = 0; j < 3; j++) {
                if (heroes[j]->isAlive()) {
                    enemies[i]->act(heroes[j]);
                    break;
                }
            }
        }

        bool heroAlive = false, enemyAlive = false;
        for (int i = 0; i < 3; i++) {
            if (heroes[i]->isAlive())  heroAlive  = true;
            if (enemies[i]->isAlive()) enemyAlive = true;
        }
        if (!heroAlive || !enemyAlive) battleActive = false;
        round++;
    }

    bool victory = false;
    for (int i = 0; i < 3; i++)
        if (heroes[i]->isAlive()) victory = true;

    if (victory) {
        cout << "=== VICTORY! ===" << endl;
        cout << "Loot: Staff of the Archmagus" << endl;
        cout << "Loot: 1500 Gold" << endl;
    } else {
        cout << "=== DEFEAT ===" << endl;
    }
    return 0;
}

THE FOUR OOP PILLARS

01 🔒

ENCAPSULATION

The Character class hides its data (name, hp, maxHp, mana, speed) behind protected visibility. External code can only read stats through controlled getters like getHp(), getMana(), and getSpeed(), while mutations like takeDamage(), heal(), and useMana() enforce safe limits — HP never goes negative, never exceeds maxHp, mana never drops below zero. The constructor body assigns each member directly, making it clear exactly what each parameter controls. This keeps the object's internal state consistent at all times.

02 🌳

INHERITANCE

All character types derive from the single abstract base Character. Hero classes Warrior, Mage, and Healer inherit shared logic (takeDamage, heal, useMana) and set their own stats by chaining up to the Character constructor. Enemy also inherits Character and adds a copy constructor that builds a named Clone — triggered by Enemy goblinClone = goblin in main(). Each subclass only writes the code unique to itself; shared behaviour lives once in the base class. This eliminates code duplication across all four concrete types.

03 ⚡

POLYMORPHISM

The pure-virtual act(Character*) in Character enables runtime dispatch: the same heroes[i]->act(target) call fires a warrior's sword strike, a mage's fireball, or an enemy's attack — depending on the object's true type. Two simple indexed loops iterate over Character* arrays and call act() identically for every character; the compiler dispatches to the right subclass method at runtime. No if/switch needed — this is polymorphism in action.

04 🎯

ABSTRACTION

Character is an abstract class — it cannot be created directly (Character c(...) would not compile) because act() is pure virtual. The pure virtual act() defines what every character must be able to do, without saying how. Each subclass fills in the details. The battle loop works entirely through Character* pointer arrays and never needs to know the specifics of any subclass. The Enemy copy constructor (Enemy goblinClone = goblin) also demonstrates how classes control how their objects are duplicated.
