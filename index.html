class insuffmana : public exception {
public:
    const char* what() const throw() { return "Not enough mana!"; }
};

template <typename T>
class lootbx {
    T itm;
public:
    lootbx(T i) : itm(i) {}
    T open() { return itm; }

class chrctr {
protected:
    string nm; int hp, mxhp, mn, spd; bool ishro;
public:
    chrctr(string n, int h, int m, int s, bool f)
        : nm(n), hp(h), mxhp(h), mn(m), spd(s), ishro(f) {}
    virtual ~chrctr() {}
    virtual void act(chrctr* tgt) = 0;
    int    getspd()   const { return spd; }
    string getnm()    const { return nm; }
    int    gethp()    const { return hp; }
    int    getmn()    const { return mn; }
    bool   getishro() const { return ishro; }
    bool   isalv()    const { return hp > 0; }
    friend ostream& operator<<(ostream& os, const chrctr& c) {
        return os << c.nm << " [HP:" << c.hp << "/" << c.mxhp << " MP:" << c.mn << "]";
    }
    void tkdmg(int d) { hp -= d; if (hp < 0) hp = 0; }
    void hl(int a)    { hp += a; if (hp > mxhp) hp = mxhp; }
    void decmn(int a) { mn -= a; if (mn < 0) mn = 0; }
};

class war : public chrctr {
public:
    war(string n) : chrctr(n, 150, 20, 10, true) {}
    void act(chrctr* tgt) override { tgt->tkdmg(25); }
    void operator+(int a) { hl(a); }
};

class mag : public chrctr {
    map<string, int> splbk;
public:
    mag(string n) : chrctr(n, 80, 100, 15, true) {
        splbk["Fireball"] = 30;
        splbk["IceStrike"] = 20;
    }
    void act(chrctr* tgt) override { cstspl("Fireball", tgt); }
    void cstspl(string s, chrctr* tgt) {
        if (splbk.find(s) == splbk.end()) return;
        if (mn < splbk[s]) throw insuffmana();
        decmn(splbk[s]);
        tgt->tkdmg(s == "Fireball" ? 40 : 25);
    }
};

class hlr : public chrctr {
public:
    hlr(string n) : chrctr(n, 90, 120, 14, true) {}
    void act(chrctr* tgt) override {
        if (mn < 25) throw insuffmana();
        decmn(25); tgt->hl(40);
    }
};

class enmy : public chrctr {
public:
    enmy(string n, int h, int m, int s) : chrctr(n, h, m, s, false) {}
    enmy(const enmy& e) : chrctr(e.nm + " (Clone)", e.hp, e.mn, e.spd, false) {}
    void act(chrctr* tgt) override { tgt->tkdmg(15); }
};

int main() {
    war p1("Arthur"); mag p2("Merlin"); hlr p3("Elara");
    enmy e1("Goblin Scout", 60, 0, 12);
    enmy e2("Orc Brute", 120, 0, 8);
    enmy e3 = e1;

    vector<chrctr*> hros = {&p1, &p2, &p3};
    vector<chrctr*> enms = {&e1, &e2, &e3};

    int rseed = 0;
    bool btlactv = true; int rnd = 1;

    while (btlactv) {
        vector<chrctr*> nxtturn;
        for (auto h : hros) if (h->isalv()) nxtturn.push_back(h);
        for (auto e : enms) if (e->isalv()) nxtturn.push_back(e);
        if (nxtturn.empty()) break;

        for (int i = 0; i < (int)nxtturn.size()-1; i++)
            for (int j = i+1; j < (int)nxtturn.size(); j++)
                if (nxtturn[i]->getspd() < nxtturn[j]->getspd())
                    swap(nxtturn[i], nxtturn[j]);

        for (auto curturn : nxtturn) {
            if (!curturn->isalv()) continue;
            bool hroalv = false, enmalv = false;
            for (auto h : hros) if (h->isalv()) hroalv = true;
            for (auto e : enms) if (e->isalv()) enmalv = true;
            if (!hroalv || !enmalv) { btlactv = false; break; }

            if (curturn->getishro()) {
                vector<chrctr*> alveanime, alvhros;
                for (auto e : enms) if (e->isalv()) alveanime.push_back(e);
                for (auto h : hros) if (h->isalv()) alvhros.push_back(h);
                int chc; cin >> chc;
                if (chc == 1) {
                    int t; cin >> t;
                    curturn->act(alveanime[t-1]);
                } else if (chc == 3 && dynamic_cast<mag*>(curturn)) {
                    string s; int t; cin >> s >> t;
                    try { dynamic_cast<mag*>(curturn)->cstspl(s, alveanime[t-1]); }
                    catch (insuffmana& ex) { cout << ex.what(); }
                } else if (chc == 3 && dynamic_cast<hlr*>(curturn)) {
                    int t; cin >> t;
                    try { curturn->act(alvhros[t-1]); }
                    catch (insuffmana& ex) { cout << ex.what(); }
                }
            } else {
                vector<chrctr*> alvhros;
                for (auto h : hros) if (h->isalv()) alvhros.push_back(h);
                int rndtgt = (rseed++) % (int)alvhros.size();
                curturn->act(alvhros[rndtgt]);
            }
        }
        rnd++;
    }

    bool vic = false;
    for (auto h : hros) if (h->isalv()) vic = true;
    if (vic) {
        lootbx<string> epcloot("Staff of the Archmagus");
        lootbx<int>    gldloot(1500);
        cout << epcloot.open() << "\n" << gldloot.open() << " Gold\n";
    }
    return 0;
}

THE FOUR OOP PILLARS

01 🔒

ENCAPSULATION

The chrctr class hides its data (nm, hp, mxhp, mn, spd) behind protected visibility. External code can only read stats through controlled getters like gethp(), getmn(), and getspd(), while mutations like tkdmg(), hl(), and decmn() enforce invariants (HP never goes negative, never exceeds mxhp). The mag class further encapsulates its splbk spell book as a private member, exposed only through cstspl() and shwspls().

02 🌳

INHERITANCE

All character types derive from the single abstract base chrctr. Hero classes war, mag, and hlr inherit shared logic (stats, tkdmg, hl, decmn) and provide their own stats via constructor chaining. enmy also inherits chrctr and adds a copy constructor that creates a named Clone — mirroring the enmy e3 = e1; line in main(). This eliminates code duplication across all five concrete types.

03 ⚡

POLYMORPHISM

The pure-virtual act(chrctr*) in chrctr enables runtime dispatch: the same curturn->act(target) call fires a warrior's broadsword swing, a mage's fireball, a healer's divine light, or an enemy attack — depending on the object's true type. dynamic_cast is used to query whether curturn is a mag* or hlr* at runtime to unlock class-specific menus. Operator overloading gives war a + operator for healing, and chrctr overloads << for pretty-printing.

04 📦

GENERIC TEMPLATES & EXCEPTIONS

The lootbx<T> template class is type-agnostic: it stores a string weapon name or an int gold amount with the same interface. The custom exception insuffmana extends std::exception and is thrown by mag::cstspl() and hlr::act() whenever MP falls short of a spell's cost — caught with a try/catch block in main() to waste the turn gracefully instead of crashing.
